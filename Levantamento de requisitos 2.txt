Criação de tabelas e diagramas

CREATE TABLE "cliente" (
	"id_cliente"	TEXT NOT NULL UNIQUE,
	"nome"	VARCHAR(100) NOT NULL,
	"telefone"	VARCHAR(15) UNIQUE,
	"email"	VARCHAR(100) UNIQUE,
	"cpf"	VARCHAR(14) NOT NULL UNIQUE,
	"endereco"	INTEGER,
	"data_cadastro"	DATETIME,
	UNIQUE("cpf"),
	UNIQUE("email"),
	PRIMARY KEY("id_cliente","cpf"),
	UNIQUE("telefone")
);

CREATE TABLE "produtos" (
	"id_produto"	INTEGER NOT NULL UNIQUE,
	"produto"	TEXT NOT NULL,
	"produto_complemento"	TEXT,
	"quantidade"	INTEGER NOT NULL,
	"disponibilidade"	INTEGER,
	"preco"	TEXT NOT NULL,
	"codigo_barras"	TEXT UNIQUE,
	PRIMARY KEY("id_produto" AUTOINCREMENT)
);

CREATE TABLE "servicos" (
	"id_servico"	INTEGER NOT NULL UNIQUE,
	"tipo_servico"	TEXT NOT NULL,
	"preco_servico"	TEXT NOT NULL,
	"descricao_servico"	TEXT,
	PRIMARY KEY("id_servico" AUTOINCREMENT)
);

CREATE TABLE "vendas" (
	"id_venda"	INTEGER NOT NULL UNIQUE,
	"numero_venda"	INTEGER NOT NULL UNIQUE,
	"status_venda"	INTEGER NOT NULL,
	"quantidade_produto"	INTEGER NOT NULL,
	PRIMARY KEY("id_venda" AUTOINCREMENT)
);
